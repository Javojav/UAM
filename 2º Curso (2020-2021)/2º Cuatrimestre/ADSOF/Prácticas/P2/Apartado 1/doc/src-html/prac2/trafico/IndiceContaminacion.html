001package prac2.trafico;
002
003/**
004 * Enumeración que contiene los tipos de índice de contaminación junto con los
005 * métodos necesarios para conocer el índice de contaminacion dado el año de
006 * compra
007 *
008 * @version 1.0 26 Feb 21
009 * @author David Teófilo Garitagoitia Romero
010 * @author Javier Fernandez de Alarcon Gervas
011 */
012public enum IndiceContaminacion {
013        A, B, C; //Posibles índices de contaminación
014
015        private static final int FECHAA = 2018; //fecha mínima indice A
016        private static final int FECHAB = 2010; //fecha mínima índice B
017
018        /**
019         * Método para conocer el indíce de contaminación de un objeto
020         * 
021         * @param anyo anyo de compra del objeto del que se desea conocer el índice de
022         *             contaminación
023         */
024        public static IndiceContaminacion getIndiceContaminacion(int anyo) {
025                if (anyo >= FECHAA)
026                        return A;
027                if (anyo >= FECHAB)
028                        return B;
029                return C;
030        }
031}
