001package prac2.trafico;
002
003/**
004 * Clase que contiene los datos de un vehiculo con métodos para obtener su
005 * número de ruedas (método abstracto), su índice de contaminación y para
006 * imprimir su información
007 *
008 * @version 1.0 25 Feb 21
009 * @author David Teófilo Garitagoitia Romero
010 * @author Javier Fernandez de Alarcon Gervas
011 */
012public abstract class Vehiculo {
013        private String modelo; // String que guarda el nombre del modelo del vehículo
014        private int anyoCompra; // Entero que contiene el año en el que fue adquirido el vehículo
015        private String matricula; // String que contiene la mátricula del vehículo
016
017        /**
018         * Constructor de la clase vehículo, con el modelo, el anyo de compra y la
019         * mátricula
020         * 
021         * @param mod       modelo del vehículo
022         * @param a         anyo de compra del vehículo
023         * @param matricula matrícula del vehículp
024         */
025        public Vehiculo(String mod, int a, String matricula) {
026                this.modelo = mod;
027                this.anyoCompra = a;
028                this.matricula = matricula;
029        }
030
031        /**
032         * Método para imprimir la clase vehículo (modelo, matricula, año, ruedas e
033         * índice de contaminación)
034         * 
035         * @return String descriprivo del vehículo
036         */
037        public String toString() {
038                return "modelo " + this.modelo + ", " + "matrícula: " + this.getMatricula() + ", fecha compra "
039                                + this.anyoCompra + ", con " + this.numeroRuedas() + " ruedas, índice:" + this.getIndiceContaminacion();
040        }
041
042        /**
043         * Método para conocer la matrícula del vehículo
044         * 
045         * @return el string con la matrícula del vehículo
046         */
047        public String getMatricula() {
048                return this.matricula;
049        }
050
051        /**
052         * Método abstracto para obtener el número de ruedas del vehículo
053         * 
054         * @return el número de ruedas del vehículo
055         */
056        public abstract int numeroRuedas();
057
058        /**
059         * Método para conocer el índice de contaminación del vehículo
060         * 
061         * @return el índice de contaminación del vehículo
062         */
063        public IndiceContaminacion getIndiceContaminacion() {
064                return IndiceContaminacion.getIndiceContaminacion(this.anyoCompra);
065        }
066}
