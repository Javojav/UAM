001package prac2.trafico;
002
003/**
004 * Clase que contiene los datos de un coche (si es diesel más la información de
005 * la clase vehículo de la que hereda) con métodos para obtener su número de
006 * ruedas, su índice de contaminación y para imprimir su información
007 *
008 * @version 1.0 26 Feb 21
009 * @author David Teófilo Garitagoitia Romero
010 * @author Javier Fernandez de Alarcon Gervas
011 */
012public class Coche extends Vehiculo {
013        private boolean diesel; // Booleano que recoge si el coche es diesel o no
014
015        /**
016         * Constructor de la clase coche, con el modelo, el anyo de compra y la
017         * mátricula (necesarios para el constructor del super, vehículo), y el booleano
018         * que definirá si es diesel
019         * 
020         * @param mod       modelo del coche
021         * @param a         anyo de compra del coche
022         * @param matricula matrícula del coche
023         * @param diesel    booleano para saber si es o no diesel
024         */
025        public Coche(String mod, int a, String matricula, boolean diesel) {
026                super(mod, a, matricula);
027                this.diesel = diesel;
028        }
029
030        /**
031         * Método para obtener el número de ruedas del coche
032         * 
033         * @return el número de ruedas del coche
034         */
035        public int numeroRuedas() {
036                return 4;
037        }
038
039        /**
040         * Método para imprimir la clase coche (si se trata de un coche diesesl junto
041         * con los datos de la clase de la que hereda)
042         * 
043         * @return String descriptivo del coche
044         */
045        public String toString() {
046                return "Coche " + (this.diesel ? "diesel" : "gasolina") + ", " + super.toString();
047        }
048
049        /**
050         * Método para conocer el índice de contaminación del coche
051         * 
052         * @return el índice de contaminación del coche
053         */
054        public IndiceContaminacion getIndiceContaminacion() {
055                if (this.diesel){
056                        return IndiceContaminacion.C;
057                }
058                return super.getIndiceContaminacion();
059        }
060
061        /**
062         * Método para obtener el booleano que define si la coche es diesel
063         * 
064         * @return true si es diesel, false en el caso contrario
065         */
066        public boolean getDiesel() {
067                return this.diesel;
068        }
069}
